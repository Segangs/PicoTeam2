cmake_minimum_required(VERSION 3.13)

# [중요] 1. 맥용 플래그가 추가되는 것을 방지 (project 명령어 전에 위치해야 함)
set(CMAKE_SYSTEM_NAME Generic)
set(CMAKE_SYSTEM_PROCESSOR arm)

# bare-metal 툴체인은 기본 컴파일러 체크에서 실행파일 링크가 실패할 수 있어
# try_compile을 정적 라이브러리 빌드로 전환합니다.
set(CMAKE_TRY_COMPILE_TARGET_TYPE STATIC_LIBRARY)

# set(CMAKE_CXX_STANDARD 17)
# set(CMAKE_C_STANDARD 17)

# [2] 컴파일러 자동 탐지 (Linux/mac 공통)
# - PATH에 arm-none-eabi 툴체인이 있으면 자동으로 사용합니다.
# - 이미 -D CMAKE_C_COMPILER=... 등으로 지정된 값은 유지합니다.
if(NOT CMAKE_C_COMPILER)
	find_program(PICO_ARM_GCC arm-none-eabi-gcc REQUIRED)
	set(CMAKE_C_COMPILER ${PICO_ARM_GCC})
endif()

if(NOT CMAKE_CXX_COMPILER)
	find_program(PICO_ARM_GXX arm-none-eabi-g++ REQUIRED)
	set(CMAKE_CXX_COMPILER ${PICO_ARM_GXX})
endif()

if(NOT CMAKE_ASM_COMPILER)
	set(CMAKE_ASM_COMPILER ${CMAKE_C_COMPILER}) # 어셈블러도 GCC 사용
endif()

# Pico 칩셋과 보드 이름 지정
set(PICO_PLATFORM rp2350-arm-s)
set(PICO_BOARD pico2_w)

# SDK 경로를 프로젝트 내부 deps/pico-sdk로 고정
set(PICO_SDK_PATH "${CMAKE_CURRENT_SOURCE_DIR}/deps/pico-sdk")

# 2. SDK 내부의 필수 파일을 불러옵니다.
# (pico-sdk/external/pico_sdk_import.cmake 파일을 루트로 복사해두면 더 편리합니다)
include(deps/pico-sdk/external/pico_sdk_import.cmake)

# project(SmartFarm_Project C CXX ASM)
project(PicoTeam2_Project C CXX ASM)


# 3. Pico SDK를 초기화합니다.
pico_sdk_init()


# 4. 각 팀원의 폴더나 src 폴더를 포함시킵니다.
# (이 기능을 통해 각 폴더 내부의 CMakeLists.txt를 읽어옵니다)

foreach(member_dir IN ITEMS src Segang Siyeong Jeyoon)
	if(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/${member_dir}/CMakeLists.txt")
		add_subdirectory(${member_dir})
	else()
		message(STATUS "Skip ${member_dir}: CMakeLists.txt not found")
	endif()
endforeach()
